الرئيسية sql ربط مع vs العلاقات

انشاء قاعدة بيانات جديده :

SQL (DDL)

CREATE DATABASE my_database;

عرض جميع قواعد البيانات :

SQL (DDL)

SHOW databases;

تحديد قاعدة البيانات :

SQL (DDL)

USE my_database;

هنا نقوم باختيار قاعدة البيانات الخاصة بنا لاجراء العمليات عليها

انشاء حدول جديد :

SQL (DDL)

    
CREATE TABLE students (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    PRIMARY KEY (id)
                    );
هنا قمنا بانشاء جدول جديد يحتوى على عمودين الاسم والايميل

تعديل في جدول :

SQL (DDL)

    
ALTER TABLE students
    ADD phone_number VARCHAR(255);
هنا قمنا باضافة عمود جديد في الجدول

اضافة بيانات الى الجدول :

SQL (DML)

    
INSERT INTO students (name, email,phone_number)
    VALUES 
    (
    "wael albahri",
    'wael@example.com',
    "777"
    );
هنا قمنا باضافة صف جديد الى الجدول

عرض جميع بيانات الجدول :

SQ (DML)

SELECT * FROM students;

سيتم عرض جميع بيانات الجدول

عرض جميع بيانات بشرط معين :

SQ (DML)

SELECT * FROM students WHERE id = 1;

سيتم عرض جميع الحقول اللتي تمتلك ال id 1

تعديل بيانات في جدول :

SQL (DML)

    
UPDATE students
    SET email = 'wael10@example.com'
    WHERE id = 1;
هنا قمنا بتعديل حقل الايميل لل id رقم 1

حذف بيانات من جدول :

SQL (DML)

DELETE FROM students WHERE id = 1;

هنا سيتم حذف الصف اللذي يملك ال id رقم 1

حذف جميع البيانات من الجدول :

SQL (DDL)

TRUNCATE TABLE students;

هنا سيتم حذف جميع البيانات دون حذف الجدول

تغيير اسم جدول :

SQL (DDL)

RENAME TABLE sudents TO clients;

عرض تفاصيل اعمدة الجدول :

SQL (DDL)

DESC clients;

عرض جميع الجداول في قاعدة البيانات الخاصة بنا :

SQL (DDL)

SHOW tables;

حذف الجدول من قاعدة البيانات:

SQL (DDL)

DROP TABLE clients;

هنا سيتم حذف الجدول مع البيانات من قاعدة البيانات

حذف قاعدة البيانات :

SQL (DDL)

DROP DATABASE my_database;

هنا سيتم حذف قاعدة البيانات كاملة بما فيها الجدوال
